#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>
#include <climits>
#include <cstdlib>
using namespace std;

int main()
{
	int n;
	cin >> n;cin.ignore();
	if (n <= 1 || n >= 100000) {
		cout << 0 << endl;
		return 0;
	}
	int ar[n];
	for (int i = 0; i < n; i++) {
		cin >> ar[i];cin.ignore();
		
		if (ar[i] <= 0 || ar[i] >= 10000000) {
			cout << 0 << endl;
			return 0;
		}
	}
	sort(ar, ar + n);
        int mindiff = INT_MAX;
        for(int i=0;i<n-1;i++){
		int diff = abs(ar[i]-ar[i+1]);
			if(diff<mindiff){
			mindiff=diff;
			}
		}
	cout <<mindiff<< endl;	
	}
