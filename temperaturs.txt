#include <iostream>
#include <string>
#include <vector>
#include <algorithm>

using namespace std;
int main()
{
	int n;
	int temp=5527;
	cin >> n; cin.ignore();
	if(n==0) {
        cout<<0<<endl;
	return 0;}
	
	for (int i = 0; i < n; i++) {
		int t; 
		cin >> t; cin.ignore();
		
	 if(abs(t)<abs(temp))
			temp = t;
		else if (abs(t) == abs(temp) && t > temp) {
			temp = t;
			
		}
		
		
	}
	

cout << temp;
	
}
